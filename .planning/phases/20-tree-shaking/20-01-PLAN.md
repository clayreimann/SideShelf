// Enable Expo SDK 54 tree shaking + inlineRequires (production-only).

// Revert: set EXPO_TREE_SHAKING=false in .env and rebuild.

  config.transformer.getTransformOptions = async () => ({
    transform: {
      experimentalImportSupport: true,
      inlineRequires: true,

  });
}
