    const userId = get().userProfile.user?.id;
    if (!userId) return;
    const ops = await dequeuePendingOps(userId); // FIFO order
    const succeededIds: string[] = [];
    for (const op of ops) {
      try {
        if (op.operationType === 'create') {
          const res = await createBookmark(op.libraryItemId, op.time, op.title ?? undefined);
          await upsertBookmark({ ...res.bookmark row shape ..., userId, syncedAt: new Date() });

        // stop draining on first failure to preserve order
        break;
