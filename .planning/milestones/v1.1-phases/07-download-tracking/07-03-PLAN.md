Create `src/lib/orphanScanner.ts` — a new module that walks the download directories and identifies files on disk with no matching DB record.

```typescript
import { logger } from "@/lib/logger";
import { downloadService } from "@/services/DownloadService";
import { getDownloadedAudioFilesWithLibraryInfo } from "@/db/helpers/localData";
import { Directory, File, Paths } from "expo-file-system";

const log = logger.forTag("OrphanScanner");

export interface OrphanFile {
  uri: string;
  filename: string;
  libraryItemId: string;
  size: number;
}

/**
 * Scan download directories for files with no matching DB record.
 * Called lazily on Storage tab focus — not on every foreground resume.
 * Skips library item directories that have an active download in progress.
 */
export async function scanForOrphanFiles(): Promise<OrphanFile[]> {
  const orphans: OrphanFile[] = [];

  // Build a Set of all known download paths (from DB)
  const downloadedFiles = await getDownloadedAudioFilesWithLibraryInfo();
  const knownPaths = new Set(downloadedFiles.map((f) => f.downloadPath));

  // Walk both documents and caches download directories
  const locations = [
    new Directory(Paths.document, "downloads"),
    new Directory(Paths.cache, "downloads"),
  ];

  for (const downloadsDir of locations) {
    if (!downloadsDir.exists) continue;

    // Each subdirectory is a library item ID
    const itemDirs = downloadsDir.list();
    for (const itemDir of itemDirs) {
      if (!(itemDir instanceof Directory)) continue;

      // Extract library item ID from directory name
      const libraryItemId = itemDir.uri.split("/").filter(Boolean).pop() ?? "";
      if (!libraryItemId) continue;

      // Skip items with active downloads — partial files are not orphans
      if (downloadService.isDownloadActive(libraryItemId)) {
        log.debug(`[OrphanScanner] Skipping active download: ${libraryItemId}`);
        continue;
      }

      const files = itemDir.list();
      for (const file of files) {
        if (!(file instanceof File)) continue;

        // If the file's URI is not in the known download paths, it is an orphan
        if (!knownPaths.has(file.uri)) {
          const filename = file.uri.split("/").filter(Boolean).pop() ?? file.uri;
          orphans.push({
            uri: file.uri,
            filename,
            libraryItemId,
            size: file.size ?? 0,
          });
          log.info(`[OrphanScanner] Orphan found: ${filename} (${libraryItemId})`);
        }
      }
    }
  }

  log.info(`[OrphanScanner] Scan complete. Found ${orphans.length} orphan file(s).`);
  return orphans;
}
```

Note on path matching: `getDownloadedAudioFilesWithLibraryInfo()` resolves paths via `resolveAppPath()` which returns absolute URIs. `file.uri` from expo-file-system 19 also returns an absolute URI. These should match directly. If paths use `file://` scheme on one side and not the other, normalize both with `decodeURIComponent` before comparing.
